#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main() {
    
    vector <ll> v;
    ll inp;
    while(cin >> inp) {
        v.push_back(inp);
    }
    sort(v.begin(), v.end());
    
    ll a = v[0], b = v[1], c = v[v.size()-1]-v[0]-v[1];
    cout << a << ' ';
    cout << b << ' ';
    cout << c;
    
    return 0;
}
