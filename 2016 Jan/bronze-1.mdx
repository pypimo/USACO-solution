#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main() {
    freopen("promote.in", "r", stdin);
    freopen("promote.out", "w", stdout);
    
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    ll A[5];
    ll a, b;
    
    for (ll i=0; i<4; i++) {
        cin >> a >> b;
        A[i] = b-a;
    }
    
    for (ll i=3; i>0; i--) {
        if (i!=3) A[i] += A[i+1];
    }
    
    for (ll i=1; i<=3; i++) {
        cout << A[i] << '\n';
    }
   
    return 0;
}
