#include <bits/stdc++.h>
#define ll long long
using namespace std;

const ll MAX_N = 11;
char A[MAX_N][MAX_N];

int main() {
    freopen("cowsignal.in", "r", stdin);
    freopen("cowsignal.out", "w", stdout);

    ios::sync_with_stdio(0);
    cin.tie(0);

    ll n, m, k;
    cin >> n >> m >> k;

    for (ll i=0; i<n; i++) {
        for (ll j=0; j<m; j++) {
            cin >> A[i][j];
        }
    }
    
    // for (ll i=0; i<n; i++) {
    //         for (ll j=0; j<m; j++) {
    //             cout << A[i][j];
    //         }
    //         cout << '\n';
    //     }
    
    ll row=0;
    for (ll i=0; i<k*n; i++) {
        for (ll j=0; j<m; j++) {
            for (ll i=0; i<k; i++) {
                cout << A[row][j];
            }
        }
        if (i%k==(k-1)) row++;
        cout << '\n';
    }

    return 0;
}
