#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main() {
    freopen("marathon.in", "r", stdin);
    freopen("marathon.out", "w", stdout);
    
    ios::sync_with_stdio(0);
    cin.tie(0);

    ll n, a, b, d=0;
    cin >> n;
    vector<pair<ll, ll>> v;

    for (ll i=0; i<n; i++) {
        cin >> a >> b;
        v.push_back({a, b});
    }

    for (ll i=1; i<n; i++) {
        d+=abs(v[i].first-v[i-1].first)+abs(v[i].second-v[i-1].second);
    }
    
    ll ct, m=INT_MAX;
    for (ll i=1; i<n-1; i++) {
        ct=d;
        ct-=abs(v[i].first-v[i-1].first)+abs(v[i].second-v[i-1].second) + abs(v[i].first-v[i+1].first)+abs(v[i].second-v[i+1].second);
        ct+=abs(v[i+1].first-v[i-1].first) + abs(v[i+1].second-v[i-1].second);
        m = min(m, ct);
        //cout << ct << '\n';
    }
    cout << m;

    return 0;
}
