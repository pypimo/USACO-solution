#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main() {
    freopen("cownomics.in", "r", stdin);
    freopen("cownomics.out", "w", stdout);
    
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    ll n, m;
    cin >> n >> m;
    string A[n+10];
    string B[n+10];
    
    for (ll i=0; i<n; i++) {
        cin >> B[i];
    }
    for (ll i=0; i<n; i++) {
        cin >> A[i];
    }
    
    ll s=0;
    for (ll i=0; i<m; i++) {
        map <char, ll> m1, m2;
        for (ll j=0; j<n; j++) {
            m1[A[j][i]]++;
        }
        for (ll j=0; j<n; j++) {
            m2[B[j][i]]++;
        }
        
        ll f=0;
        for (auto x: m1) {
            if (m2[x.first]) f=1;
        }
        if (!f) s++; 
    }
    cout << s;
    
    return 0;
}
