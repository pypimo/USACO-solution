#include <bits/stdc++.h>
#define ll long long
using namespace std;

double A[100010], small[100010];

int main() {
    freopen("homework.in", "r", stdin);
    freopen("homework.out", "w", stdout);
    
    ios::sync_with_stdio(0);
    cin.tie(0);

    ll n;
    double s=0, x, M=1e5;
    cin >>n;

    for (ll i=1; i<=n; i++) {
        cin >> A[i];
    }
    for (ll i=n; i>0; i--) {
        M=min(M, A[i]);
        small[i]=M;
    }
    for (ll i=1; i<=n; i++) {
        A[i]+=A[i-1];
    }
    for (ll i=1; i<=n-2; i++) {
        x = (A[n]-A[i]-small[i+1])/(n-i-1ll);
        s = max(s, x);
    }
    ll f=0;
    for (ll i=1; i<=n-2; i++) {
        x = (A[n]-A[i]-small[i+1])/(n-(ll)i-1ll);
        if (x==s) cout <<i << '\n';
    }

    return 0;
}
