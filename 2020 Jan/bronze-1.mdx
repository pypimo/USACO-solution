#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main() {
    freopen("word.in", "r", stdin);
    freopen("word.out", "w", stdout);
    
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    ll n, k;
    cin >> n >> k;
    string A[n+10];
    
    for (ll i=0; i<n; i++) {
        cin >> A[i];
    }
    
    ll ct = A[0].length();
    cout << A[0];
    
    for (ll i=1; i<n; i++) {
        ct += A[i].length();
        
        if (ct>k) {
            cout << '\n' << A[i];
            ct = A[i].length();
        } else {
            cout << ' ' << A[i];
        }
    }
    
    return 0;
}
