#include <bits/stdc++.h>
#define ll long long
using namespace std;

const ll MAX_N = 4;
ll M[MAX_N], C[MAX_N];

int main() {
    freopen("mixmilk.in", "r", stdin);
    freopen("mixmilk.out", "w", stdout);

    ios::sync_with_stdio(0);
    cin.tie(0);

    ll x;
    for (ll i=0; i<=2; i++) {
        cin >> C[i] >> M[i];
    }

    for (ll i=0; i<=2; i++) {
        x = min(C[(i+1)%3]-M[(i+1)%3], M[i]);
        M[i]-=x;
        M[(i+1)%3]+=x;
    }
    
    x = min(C[1]-M[1], M[0]);
    M[0]-=x;
    M[1]+=x;
        
    for (ll i=0; i<=2; i++) {
        cout << M[i] << '\n';
    }

    return 0;
}
