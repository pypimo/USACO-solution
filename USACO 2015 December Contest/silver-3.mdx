## Problem-3(SILVER) BREED COUNTING
http://www.usaco.org/index.php?page=viewproblem2&cpid=572

```cpp
#include <bits/stdc++.h>
#define ll long long
using namespace std;

ll h[100010], g[100010], j[100010];

int main() {
    freopen("bcount.in", "r", stdin);
    freopen("bcount.out", "w", stdout);
    ios::sync_with_stdio(0);
    cin.tie(0);

    ll n, q, inp, a, b;
    cin >> n >> q;

    for (ll i=1; i<=n; i++) {
        cin >> inp;
        if (inp==1) h[i]=1;
        if (inp==2) g[i]=1;
        if (inp==3) j[i]=1;
    }

    for (ll i=1; i<=n; i++) {
        h[i]+=h[i-1];
        g[i]+=g[i-1];
        j[i]+=j[i-1];
    }

    while(q--) {
        cin >> a >> b;
        cout << h[b]-h[a-1] << " "
        << g[b]-g[a-1] << " "
        << j[b]-j[a-1] << '\n';
    }
    return 0;
}
```
