#include <bits/stdc++.h>
#define ll long long
using namespace std;

vector<ll> adj[110];

void dfs(ll node, ll vis[]) {
    vis[node]=1;
    
    for (auto next: adj[node]) {
        if (!vis[next]) dfs(next, vis);
    }
    return;
}

int main() {
    freopen("factory.in", "r", stdin);
    freopen("factory.out", "w", stdout);
    
    ios::sync_with_stdio(0);
    cin.tie(0); 
    
    ll n, a, b;
    cin >> n;
    
    for (ll i=1; i<n; i++) {
        cin >> a >> b;
        adj[b].push_back(a);
    }
    
    for (ll i=1; i<=n; i++) {
        ll vis[110]={0};  
        dfs(i, vis);
        
        ll f=1;
        for (ll j=1; j<=n; j++) {
            if (!vis[j]) f=0;
        }
        if (f) {cout << i; return 0;}
    }
    
    cout << "-1";
    
    return 0;
}
