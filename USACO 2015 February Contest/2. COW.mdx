## PROBLEM-2  COW

http://www.usaco.org/index.php?page=viewproblem2&cpid=527

```cpp
#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main() {
    freopen("cow.in", "r", stdin);
	freopen("cow.out", "w", stdout);

    ios::sync_with_stdio(0);
    cin.tie(0);

    ll n,C=0, O=0, W=0;
    cin >> n;

    string s;
    cin >> s;

    for (ll i=0; i<n; i++) {
        if (s[i]=='C') C++;
        if (s[i]=='O') O+=C;
        if (s[i]=='W') W+=O;
    }
    cout << W;
    return 0;
}
```
